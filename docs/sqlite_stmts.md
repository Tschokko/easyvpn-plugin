CREATE TABLE VPN_CLIENTS (ID INTEGER PRIMARY KEY AUTOINCREMENT, CN TEXT, 
    IS_ACTIVE INTEGER NOT NULL DEFAULT(0), IPV4_ADDR TEXT NOT NULL, 
    IPV4_REMOTE_ADDR NOT NULL, IPV6_ADDR TEXT, IPV6_REMOTE_ADDR TEXT);
